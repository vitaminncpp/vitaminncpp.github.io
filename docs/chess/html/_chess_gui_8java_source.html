142 public void setAnnotatted() {
143 int x = ChessGui.this.state.getChXPrev();
144 int y = ChessGui.this.state.getChYPrev();
145 if (ChessGui.this.move == null) {
146 return;
147 }
148 switch (ChessGui.this.move.getState()) {
149 case INVALID_SELECTION:
150 break;
151 case NORMAL_MOVE:
152 this.grid[ChessGui.this.move.getX1()][ChessGui.this.move.getY1()].setSourceTile();
153 this.grid[ChessGui.this.move.getX2()][ChessGui.this.move.getY2()].setDestinationTile();
154 break;
155 case CAPTURE_MOVE:
156 this.grid[ChessGui.this.move.getX1()][ChessGui.this.move.getY1()].setSourceTile();
157 this.grid[ChessGui.this.move.getX2()][ChessGui.this.move.getY2()].setThreatenTile();
158 default:
159 break;
160 }
161 switch (state.getCheckState()) {
162 case WHITE_CHECKMATE:
163 this.grid[ChessGui.this.game.getPlayerWhite().getPieces()[5][0].getX()][ChessGui.this.game.getPlayerWhite().getPieces()[5][0].getY()].setCheckMateTile();
164 break;
165 case BLACK_CHECKMATE:
166 this.grid[ChessGui.this.game.getPlayerBlack().getPieces()[5][0].getX()][ChessGui.this.game.getPlayerBlack().getPieces()[5][0].getY()].setCheckMateTile();
167 break;
168 case WHITE_CHECK:
169 this.grid[ChessGui.this.game.getPlayerWhite().getPieces()[5][0].getX()][ChessGui.this.game.getPlayerWhite().getPieces()[5][0].getY()].setCheckTile();
170 break;
171 case BLACK_CHECK:
172 this.grid[ChessGui.this.game.getPlayerBlack().getPieces()[5][0].getX()][ChessGui.this.game.getPlayerBlack().getPieces()[5][0].getY()].setCheckTile();
173 break;
174 case WHITE_STALEMATE:
175 break;
176 case BLACK_STALEMATE:
177 break;
178
179 default:
180 break;
181 }
182 switch (ChessGui.this.state.getState()) {
183 case SELECTED_STATE:
184 this.grid[ChessGui.this.state.getChXPrev()][ChessGui.this.state.getChYPrev()].setSelectTile();
185 for (int i = 0; i < 8; i++) {
186 for (int j = 0; j < 8; j++) {
187 if (ChessGui.this.game.getPiece(ChessGui.this.state.getChXPrev(), ChessGui.this.state.getChYPrev()).getMoveMap()[i][j]) {
188 this.grid[i][j].setPossibleTile();
189 }
190 }
191 }
192 break;
193 case EMPTY_SELECTION:
194 this.grid[ChessGui.this.state.getChXPrev()][ChessGui.this.state.getChYPrev()].setEmptyTile();
195 break;
196 case INVALID_SELECTION:
197 this.grid[ChessGui.this.state.getChXPrev()][ChessGui.this.state.getChYPrev()].setInvalidTile();

322 ChessGui.this.state.reset();
