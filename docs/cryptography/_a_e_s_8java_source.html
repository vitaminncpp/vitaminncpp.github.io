61 private static byte[] cipherData(PaddedBufferedBlockCipher cipher, byte[] data) throws Exception {
62 int minSize = cipher.getOutputSize(data.length);
63 byte[] outBuf = new byte[minSize];
64 int length1 = cipher.processBytes(data, 0, data.length, outBuf, 0);
65 int length2 = cipher.doFinal(outBuf, length1);
66 int actualLength = length1 + length2;
67 byte[] result = new byte[actualLength];
68 System.arraycopy(outBuf, 0, result, 0, result.length);
