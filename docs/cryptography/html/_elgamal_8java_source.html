184 public static IvParameterSpec createCtrIvForAES(int messageNumber, SecureRandom random) {
185 byte[] ivBytes = new byte[16];
186 // initially randomize
187 random.nextBytes(ivBytes);
188 // set the message number bytes
189 ivBytes[0] = (byte) (messageNumber >> 24);
190 ivBytes[1] = (byte) (messageNumber >> 16);
191 ivBytes[2] = (byte) (messageNumber >> 8);
192 ivBytes[3] = (byte) (messageNumber >> 0);
193 // set the counter bytes to 1
194 for (int i = 0; i != 7; i++) {
195 ivBytes[8 + i] = 0;
196 }
197 ivBytes[15] = 1;
198 return new IvParameterSpec(ivBytes);
199 }
