231
232 // determine the N length parameter:
233 int N = key.length / 4; // consider only the first 4*N bytes
234 if (N < 4 || N > 10) {
235 throw new RuntimeException("Invalid Anubis key size: " + (32 * N) + " bits.");
236 }
237 int[] kappa = new int[N];
238 int[] inter = new int[N];
239
240 // determine number of rounds from key size:
241 int R = 8 + N;
242 roundKeyEnc = new int[R + 1][4];
243 roundKeyDec = new int[R + 1][4];
244
245 // map byte array cipher key to initial key state (mu):
246 for (int i = 0, pos = 0; i < N; i++) {
247 kappa[i] = ((key[pos++]) << 24) ^ ((key[pos++] & 0xff) << 16) ^ ((key[pos++] & 0xff) << 8)
248 ^ ((key[pos++] & 0xff));
249 }
250
251 // generate R + 1 round keys:
252 for (int r = 0; r <= R; r++) {
253 int K0, K1, K2, K3;
254 /*
255 * generate r-th round key K^r:

258 K1 = T4[(kappa[N - 1] >>> 16) & 0xff];
259 K2 = T4[(kappa[N - 1] >>> 8) & 0xff];
260 K3 = T4[(kappa[N - 1]) & 0xff];

263 ^ (T5[(K0 >>> 8) & 0xff] & 0x0000ff00) ^ (T5[(K0) & 0xff] & 0x000000ff);

312 protected final void crypt(byte[/* 16 */] block, int[/* R + 1 */][/* 4 */] roundKey) {
313 int[] state = new int[4];
314 int[] inter = new int[4];
315 int R = roundKey.length - 1; // number of rounds
316

318 * map byte array block to cipher state (mu) and add initial round key

320 */
321 for (int i = 0, pos = 0; i < 4; i++) {
322 state[i] = ((block[pos++]) << 24) ^ ((block[pos++] & 0xff) << 16) ^ ((block[pos++] & 0xff) << 8)
323 ^ ((block[pos++] & 0xff)) ^ roundKey[0][i];
324 }
325
326 // R - 1 full rounds:

328 inter[0] = T0[(state[0] >>> 24)] ^ T1[(state[1] >>> 24)] ^ T2[(state[2] >>> 24)] ^ T3[(state[3] >>> 24)]

334 inter[3] = T0[(state[0]) & 0xff] ^ T1[(state[1]) & 0xff] ^ T2[(state[2]) & 0xff] ^ T3[(state[3]) & 0xff]

338 }
339 }
340

343 */

345 ^ (T2[(state[2] >>> 24)] & 0x0000ff00) ^ (T3[(state[3] >>> 24)] & 0x000000ff) ^ roundKey[R][0];
